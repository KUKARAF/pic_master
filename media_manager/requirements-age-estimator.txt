# Dependencies for the experimental age/gender estimator (MiVOLO) — see
# media_manager/age_estimator.py and age_estimator_worker.py.
#
# These MUST be installed into a SEPARATE, isolated virtualenv (not the main app's
# .venv): MiVOLO hard-pins an old ultralytics/timm that conflict with this app's own
# YOLO-World object detector and CLIP indexer (confirmed: installing mivolo into the
# main venv broke `from ultralytics import YOLOWorld`).
#
# Preferred setup: `media age-setup` — creates the venv at
# ~/.local/share/media_manager/age-venv and installs this file into it (see
# setup_age_venv in age_estimator.py). Manual equivalent from a repo checkout:
#   uv venv --python 3.12 .age-venv
#   uv pip install --python .age-venv/bin/python 'setuptools<81' wheel
#   uv pip install --python .age-venv/bin/python --no-build-isolation-package mivolo \
#       -r media_manager/requirements-age-estimator.txt
# Python 3.12 because the pinned torch==2.5.1 has no wheels for 3.14+; the
# setuptools<81 + no-build-isolation dance because MiVOLO's setup.py imports
# pkg_resources, which modern setuptools no longer ships.
#
# torch/torchvision are pinned older than the main app's versions because MiVOLO's
# own model-construction code (mivolo.model.create_timm_model) fails to import
# against current timm, and current torch defaults torch.load's weights_only=True
# in a way old ultralytics (which MiVOLO pins) wasn't updated to handle. Pinning an
# older, matched torch/torchvision pair sidesteps both issues without touching any
# security default.
torch==2.5.1
torchvision==0.20.1
opencv-python
numpy
transformers
accelerate
mivolo @ git+https://github.com/WildChlamydia/MiVOLO.git
